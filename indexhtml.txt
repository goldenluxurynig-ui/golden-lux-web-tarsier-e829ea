  {name:"Pimple Patch", price:1000},
  {name:"Phone Suction", price:1300},
  {name:"Mini Fan", price:3800},
  {name:"Tote Bag", price:2000}
];

let cart = [];

function loadProducts(){
  let box = document.getElementById("products");
  products.forEach(p=>{
    let div = document.createElement("div");
    div.className="card";
    div.innerHTML = `
      <h3>${p.name}</h3>
      <p>₦${p.price}</p>
      <button onclick="add('${p.name}',${p.price})">Add to Cart</button>
    `;
    box.appendChild(div);
  });
}
loadProducts();

function add(name,price){
  let item = cart.find(i => i.name === name);
  if(item){
    item.qty += 1;
  } else {
    cart.push({name, price, qty:1});
  }
  updateCart();
}

function removeItem(name){
  cart = cart.filter(i => i.name !== name);
  updateCart();
}

function updateCart(){
  let list=document.getElementById("cartList");
  let total=0;
  list.innerHTML="";

  cart.forEach(i=>{
    let li=document.createElement("li");
    li.innerHTML = `
      ${i.name} x${i.qty} - ₦${i.price * i.qty}
      <button onclick="removeItem('${i.name}')">❌</button>
    `;
    list.appendChild(li);
    total += i.price * i.qty;
  });

  document.getElementById("total").innerText = total;
}

/* ---------------- PAYMENT ---------------- */
function pay(){
  let total = cart.reduce((a,b)=>a + (b.price * b.qty), 0);

  if(total === 0){
    alert("Cart is empty");
    return;
  }

  let email = prompt("Enter your email");

  let handler = PaystackPop.setup({
    key: "pk_test_a23deac67faa41f40a243f9923d26928fa9e0fe2",
    email: email,
    amount: total * 100,
    currency: "NGN",
    callback: async function(response){

      // Save order to Supabase (optional but ready)
      const { error } = await supabaseClient.from("orders").insert([
        {
          email: email,
          items: cart,
          total: total,
          payment_ref: response.reference
        }
      ]);

      if(error){
        alert("Payment successful but not saved");
        console.log(error);
      } else {
        alert("Payment Successful 💖 Order saved!");
      }

      cart = [];
      updateCart();
    }
  });

  handler.openIframe();
}
